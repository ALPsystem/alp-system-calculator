<script>

    function calculateALP() {

        const ptsI = parseFloat(document.getElementById('ptsIndex').value) || 0;

        const ptsC = parseFloat(document.getElementById('ptsContra').value) || 0;

        const sattI = parseFloat(document.getElementById('sattIndex').value) || 0;

        const sattC = parseFloat(document.getElementById('sattContra').value) || 0;

        const chronic = document.getElementById('isChronic').checked;

        const laxity = document.getElementById('isLaxity').checked;


        // Metrics

        const rawDeltaSatt = sattI - sattC;

        const deltaPts = ptsI - ptsC;

        const adjDeltaSatt = rawDeltaSatt - (deltaPts * 1.2);


        document.getElementById('valRawSatt').innerText = rawDeltaSatt.toFixed(1) + '%';

        document.getElementById('valDeltaPts').innerText = deltaPts.toFixed(1) + '°';

        document.getElementById('valAdjSatt').innerText = adjDeltaSatt.toFixed(1) + '%';


        // Logic

        const steepBone = ptsI >= 12;

        const pathologicalShift = adjDeltaSatt >= 7;

        const benignShift = adjDeltaSatt <= 3;


        let phenotype = "Unclassified / Transition Zone";

        let indication = "Clinical correlation required.";


        if (steepBone && pathologicalShift && chronic && laxity) {

            phenotype = "The Triple Hit (Catastrophic)";

            indication = "Indication: Critical";

        } else if (steepBone && pathologicalShift && chronic && !laxity) {

            phenotype = "Double Hit A (Synergistic Failure)";

            indication = "Indication: High";

        } else if (steepBone && pathologicalShift && !chronic && laxity) {

            phenotype = "Double Hit C (Compounded Risk)";

            indication = "Indication: Moderate to High";

        } else if (!steepBone && pathologicalShift && !chronic && laxity) {

            phenotype = "Double Hit B (Acute-on-Baseline)";

            indication = "Indication: Moderate";

        } else if (steepBone && pathologicalShift && !chronic && !laxity) {

            phenotype = "Type 1: Osseous (PTS-Driven)";

            indication = "Indication: Moderate to High";

        } else if (!steepBone && pathologicalShift && chronic && !laxity) {

            phenotype = "Type 2: Acquired Soft-Tissue";

            indication = "Indication: Moderate to High";

        } else if (laxity && benignShift) {

            phenotype = "Type 3: Inherent Soft-Tissue (Hyperlaxity)";

            indication = "Indication: Low / Contraindicated";

        }


        document.getElementById('resultPhenotype').innerText = phenotype;

        document.getElementById('resultIndication').innerText = indication;

    }


    // Initialize on load

    calculateALP();

</script>
